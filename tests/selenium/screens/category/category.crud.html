Создать, изменить, удалить категорию с HTML-символами
open /info/
clickAndWait link=Категории
click add_category
click namecat
type namecat категория с < "кавычками" >
click btnSave
waitForAjaxRequests 3000
assertTextPresent категория с < "кавычками" >
click //a[.='категория с < "кавычками" >']/../div[@class='cont']/ul/li[1]
type namecat категория & с < "кавычками" >
click btnSave
waitForAjaxRequests 3000
assertTextPresent категория & с < "кавычками" >
click //a[.='категория & с < "кавычками" >']/../div[@class='cont']/ul/li[2]
assertConfirmation Удалить категорию?
waitForAjaxRequests 3000
assertTextNotPresent категория & с < "кавычками" >
